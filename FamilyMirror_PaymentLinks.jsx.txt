function UploadBox({ label, emoji, imageData, onImage, required }) {
  const ref = useRef();
  const handleFile = (e) => {
    const file = e.target.files[0]; if (!file) return;
    const reader = new FileReader();
    reader.onload = (ev) => onImage(ev.target.result);
    reader.readAsDataURL(file);
  };
  return (
    <div className={`upload-box ${imageData?"uploaded":""} ${required&&!imageData?"required":""}`} onClick={() => ref.current.click()}>
      <input ref={ref} type="file" accept="image/*" onChange={handleFile} style={{ display:"none" }}/>
      {imageData
        ? <><img src={imageData} alt={label} style={{ width:60,height:60,borderRadius:14,objectFit:"cover",border:"2px solid #14B8A6" }}/><div style={{ fontSize:10,color:C.teal,marginTop:5,fontWeight:700 }}>✓ הועלה</div></>
        : <><div style={{ fontSize:30,marginBottom:4 }}>{emoji}</div><div style={{ fontSize:11,color:C.muted,fontWeight:600 }}>{label}</div>{required&&<div style={{ fontSize:9,color:C.pink,marginTop:2 }}>* חובה</div>}</>}
    </div>
  );
}

function AnimatedBar({ pct, color, delay=0 }) {
  const [w, setW] = useState(0);
  useEffect(() => { const t=setTimeout(()=>setW(pct),delay+100); return ()=>clearTimeout(t); },[pct,delay]);
  return (
    <div style={{ background:"rgba(255,255,255,0.07)", borderRadius:99, height:9, overflow:"hidden" }}>
      <div style={{ width:w+"%", height:"100%", background:`linear-gradient(90deg,${color},${color}88)`, borderRadius:99, transition:"width 1s cubic-bezier(0.34,1.56,0.64,1)" }}/>
    </div>
  );
}

function CountUp({ target, duration=1200 }) {
  const [val, setVal] = useState(0);
  useEffect(() => {
    let start=null;
    const step=(ts)=>{ if(!start)start=ts; const p=Math.min((ts-start)/duration,1); setVal(Math.round(p*target)); if(p<1)requestAnimationFrame(step); };
    requestAnimationFrame(step);
  },[target,duration]);
  return <span>{val}%</span>;
}

export default function App() {
  const [screen, setScreen] = useState(SCREENS.HOME);
  const [images, setImages] = useState({baby:null,mom:null,dad:null,gpa:null,gma:null});
  const [result, setResult] = useState(null);
  const [paid, setPaid] = useState(false);
  const go=(sc)=>setScreen(sc);
  const noNav=[SCREENS.ANALYZING];

  const render=()=>{
    switch(screen){
      case SCREENS.HOME:      return <HomeScreen go={go} paid={paid}/>;
      case SCREENS.UPLOAD:    return <UploadScreen go={go} images={images} setImages={setImages}/>;
      case SCREENS.ANALYZING: return <AnalyzingScreen go={go} images={images} setResult={setResult}/>;
      case SCREENS.RESULTS:   return <ResultsScreen go={go} result={result} images={images} paid={paid} setPaid={setPaid}/>;
      case SCREENS.TREE:      return <TreeScreen go={go} images={images} result={result}/>;
      case SCREENS.PAYMENT:   return <PaymentScreen go={go} setPaid={setPaid}/>;
      case SCREENS.SHARE:     return <ShareScreen go={go} result={result} images={images}/>;
      default:                return <HomeScreen go={go} paid={paid}/>;
    }
  };

  return (
    <div style={{ direction:"rtl",fontFamily:"'Plus Jakarta Sans','Segoe UI',Arial,sans-serif",minHeight:"100vh",background:`linear-gradient(160deg,${C.bg0} 0%,${C.bg1} 50%,${C.bg2} 100%)`,color:C.white,maxWidth:430,margin:"0 auto",position:"relative" }}>
      <StyleInjector/>
      {render()}
      {!noNav.includes(screen)&&<Nav screen={screen} go={go}/>}
    </div>
  );
}
